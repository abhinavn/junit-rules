17  package com.btmatthews.junit.rules;
18  
19  import org.junit.rules.TestRule;
20  import org.junit.runner.Description;
21  import org.junit.runners.model.Statement;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.text.MessageFormat;
26  import java.util.ResourceBundle;
27  
28  /**
29   * This JUnit rule generates log messages to delineate the start and finish of a JUnit test case and also to note any exceptions
30   * that are thrown.
31   *
32   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
33   * @version 1.0.0
34   */
35  public class LoggingRule implements TestRule {
36      /**
37       * The key for the message used to log the start of the test case.
38       */
39      private static final String START = "com.btmatthews.junit.rules.logging.START";
40      /**
41       * The key for the message used to log any exception thrown by the test case.
42       */
43      private static final String EXCEPTION = "com.btmatthews.junit.rules.logging.EXCEPTION";
44      /**
45       * The key for the message used to log the finish of the test case.
46       */
47      private static final String FINISH = "com.btmatthews.junit.rules.logging.FINISH";
48      /**
49       * The resource bundle from which the log messages are loaded.
50       */
51      private static final ResourceBundle messages = ResourceBundle.getBundle("com.btmatthews.junit.rules.messages");
52  
53      /**
54       * Apply the test rule by building a wrapper {@link Statement} that logs a messages before and after evaluating
55       * <code>statement</code> and if
56       * @param statement
57       * @param description
58       * @return
59       */
60      @Override
61      public Statement apply(final Statement statement, final Description description) {
62          return new Statement() {
63              @Override
64              public void evaluate() throws Throwable {
65                  final Logger logger = LoggerFactory.getLogger(description.getTestClass());
66                  logger.info(MessageFormat.format(messages.getString(START), description.getDisplayName()));
67                  try {
68                      statement.evaluate();
69                  } catch (final Throwable e) {
70                      logger.info(MessageFormat.format(messages.getString(EXCEPTION), description.getDisplayName()), e);
71                      throw e;
72                  } finally {
73                      logger.info(MessageFormat.format(messages.getString(FINISH), description.getDisplayName()));
74                  }
75              }
76          };
77      }
78  }
